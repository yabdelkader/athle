SELECT l.idEpreuve, l.nom, l.prenom, l.perf, l.annee, (l.annee - l.naissance) as age from ligne l WHERE 
l.club like 'us creteil' and (l.annee - l.naissance) > 34 and (l.annee - l.naissance) < 40 and l.sexe = 1
AND perf  IN 
	(SELECT min(perf) 
	FROM ligne l2
	WHERE l2.club like 'us creteil' and (l2.annee - l2.naissance) > 34 and (l2.annee - l2.naissance) < 40 and l2.sexe = 1
	GROUP BY l2.idEpreuve
	)
